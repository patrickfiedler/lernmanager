const topicSlug = '{{ task.name|slugify }}';
const subtaskCount = {{ subtasks|length }};

function toggleSubtask(position, checked) {
    const csrfToken = document.querySelector('meta[name="csrf-token"]').content;

    // Disable checkbox during request
    const checkboxes = document.querySelectorAll('.subtask-checkbox');
    checkboxes.forEach(cb => cb.disabled = true);

    fetch(`/schueler/thema/${topicSlug}/aufgabe/${position}`, {
        method: 'POST',
        headers: {
            'Content-Type': 'application/json',
            'X-CSRFToken': csrfToken
        },
        body: JSON.stringify({erledigt: checked})
    })
    .then(response => {
        if (!response.ok) {
            throw new Error(`HTTP error! status: ${response.status}`);
        }
        return response.json();
    })
    .then(data => {
        if (data.show_quiz && data.quiz_url) {
            window.location.href = data.quiz_url;
            return;
        }

        if (checked) {
            const successBanner = document.getElementById('success-banner');
            if (successBanner) {
                successBanner.style.display = 'flex';
            }

            // Update progress dot
            const currentDot = document.querySelector('.progress-dots .dot.current');
            if (currentDot) {
                currentDot.classList.add('completed');
            }

            // Update quiz dot so goToNextSubtask() knows the quiz is now available
            const quizDot = document.querySelector(`.dot-subtask-quiz[data-quiz-position="${position}"]`);
            if (quizDot && quizDot.dataset.quizPassed !== 'true') {
                quizDot.dataset.quizAvailable = 'true';
                quizDot.classList.add('available');
            }

            updateProgressText();
        } else {
            // Unchecking - reload page to reset state
            setTimeout(() => {
                window.location.href = window.location.href;
            }, 100);
        }
    })
    .catch(err => {
        console.error('Error toggling subtask:', err);
        checkboxes.forEach(cb => cb.disabled = false);
        alert(`Fehler beim Speichern: ${err.message}\nBitte versuche es erneut oder lade die Seite neu.`);
    });
}

function updateProgressText() {
    const allDots = document.querySelectorAll('.progress-dots .dot.dot-subtask');
    const requiredDots = Array.from(allDots).filter(d => d.dataset.required !== 'false');
    const completedRequired = requiredDots.filter(d => d.classList.contains('completed'));
    const progressText = document.getElementById('progress-text');

    if (progressText) {
        progressText.textContent = `${completedRequired.length} von ${requiredDots.length} Aufgaben erledigt`;
    }
}

function goToNextSubtask(currentIndex) {
    const position = currentIndex + 1; // 1-based
    // Check if current task has an unpassed quiz → go there first
    const quizDot = document.querySelector(`.dot-subtask-quiz[data-quiz-position="${position}"]`);
    if (quizDot && quizDot.dataset.quizPassed !== 'true' && quizDot.dataset.quizAvailable === 'true') {
        navigateToSubtaskQuiz(position, false, true);
        return;
    }
    if (currentIndex < subtaskCount - 1) {
        const url = new URL(window.location);
        url.searchParams.set('aufgabe', currentIndex + 2);
        window.location.href = url.toString();
    } else {
        const topicQuizDot = document.querySelector('.dot-quiz');
        if (topicQuizDot) {
            navigateToQuiz(false);
        }
    }
}

function goToPrevSubtask(currentIndex) {
    if (currentIndex > 0) {
        const url = new URL(window.location);
        url.searchParams.set('aufgabe', currentIndex);
        window.location.href = url.toString();
    }
}

function navigateToSubtask(position) {
    const url = new URL(window.location);
    url.searchParams.set('aufgabe', position);
    window.location.href = url.toString();
}

function navigateToQuiz(passed) {
    if (passed) {
        window.location.href = `/schueler/thema/${topicSlug}/quiz-ergebnis`;
    } else {
        window.location.href = `/schueler/thema/${topicSlug}/quiz`;
    }
}

function navigateToSubtaskQuiz(position, passed, available) {
    if (passed) {
        window.location.href = `/schueler/thema/${topicSlug}/aufgabe-${position}/quiz-ergebnis`;
    } else if (available) {
        window.location.href = `/schueler/thema/${topicSlug}/aufgabe-${position}/quiz`;
    }
    // locked (not available) — do nothing
}

// Keyboard navigation for all dots (WCAG 2.1)
function handleDotKeydown(event) {
    if (event.key === 'Enter' || event.key === ' ') {
        event.preventDefault();
        event.target.click();
    } else if (event.key === 'ArrowLeft' || event.key === 'ArrowRight') {
        event.preventDefault();
        const dots = Array.from(document.querySelectorAll('.progress-dots .dot'));
        const currentIndex = dots.indexOf(event.target);

        let nextIndex;
        if (event.key === 'ArrowLeft') {
            nextIndex = currentIndex > 0 ? currentIndex - 1 : dots.length - 1;
        } else {
            nextIndex = currentIndex < dots.length - 1 ? currentIndex + 1 : 0;
        }

        dots[nextIndex].focus();
    }
}

// --- Artifact upload flow ---

let artifactPreviewText = '';

function handleArtifactUpload(position) {
    const fileInput = document.getElementById('artifact-file-input');
    const file = fileInput.files[0];
    if (!file) return;

    document.getElementById('artifact-filename').textContent = file.name;
    document.getElementById('artifact-status').textContent = 'Wird gelesen …';
    const feedbackBtn = document.getElementById('artifact-feedback-btn');
    feedbackBtn.style.display = 'none';
    feedbackBtn.textContent = 'KI-Check starten →';
    feedbackBtn.disabled = false;
    document.getElementById('artifact-checklist').style.display = 'none';
    document.getElementById('artifact-preview-details').style.display = 'none';

    const csrfToken = document.querySelector('meta[name="csrf-token"]').content;
    const formData = new FormData();
    formData.append('file', file);

    fetch(`/schueler/thema/${topicSlug}/aufgabe-${position}/artefakt/vorschau`, {
        method: 'POST',
        headers: {'X-CSRFToken': csrfToken},
        body: formData,
    })
    .then(r => r.json())
    .then(data => {
        if (data.error) {
            document.getElementById('artifact-status').textContent = '⚠️ ' + data.error;
            return;
        }
        artifactPreviewText = data.preview_text;
        document.getElementById('artifact-preview-text').textContent = data.preview_text;
        document.getElementById('artifact-preview-filename').textContent = data.filename;
        document.getElementById('artifact-preview-details').style.display = '';
        document.getElementById('artifact-preview-details').open = true;
        document.getElementById('artifact-feedback-btn').style.display = '';
        document.getElementById('artifact-status').textContent = '';
    })
    .catch(() => {
        document.getElementById('artifact-status').textContent = '⚠️ Fehler beim Lesen der Datei.';
    });
}

function requestArtifactFeedback(position) {
    const btn = document.getElementById('artifact-feedback-btn');
    const statusEl = document.getElementById('artifact-status');
    btn.disabled = true;
    btn.textContent = '⏳ KI prüft …';
    statusEl.textContent = '';

    const csrfToken = document.querySelector('meta[name="csrf-token"]').content;
    fetch(`/schueler/thema/${topicSlug}/aufgabe-${position}/artefakt/feedback`, {
        method: 'POST',
        headers: {'Content-Type': 'application/json', 'X-CSRFToken': csrfToken},
        body: JSON.stringify({text: artifactPreviewText}),
    })
    .then(r => r.json())
    .then(data => {
        statusEl.textContent = '';
        if (data.rate_limited) {
            btn.textContent = 'KI-Check starten →';
            btn.disabled = true;
            statusEl.innerHTML = '⏰ Du hast heute schon 10 Mal geprüft. Warte eine Stunde — dann kannst du wieder prüfen.';
            return;
        }
        btn.textContent = 'Erneut prüfen →';
        btn.disabled = false;
        if (data.error) {
            statusEl.textContent = '⚠️ ' + data.error;
            return;
        }
        if (data.llm_disabled) {
            statusEl.textContent = 'ℹ️ KI-Feedback ist für diese Klasse nicht aktiviert.';
            return;
        }
        if (data.checks_remaining !== undefined && data.checks_remaining <= 3 && data.checks_remaining > 0) {
            statusEl.textContent = `Noch ${data.checks_remaining} Check${data.checks_remaining === 1 ? '' : 's'} in dieser Stunde übrig.`;
        }
        renderArtifactChecklist(data.feedback);
    })
    .catch(() => {
        btn.textContent = 'KI-Check starten →';
        btn.disabled = false;
        statusEl.textContent = '⚠️ Verbindungsfehler. Bitte nochmal versuchen.';
    });
}

function renderArtifactChecklist(feedback) {
    const container = document.getElementById('artifact-checklist');
    if (!feedback || feedback.length === 0) {
        container.innerHTML = '
⚠️ KI konnte die Datei nicht auswerten. Bitte nochmal versuchen.
';
        container.style.display = '';
        return;
    }
    const items = feedback.map(item => {
        const icon = item.passed ? '✅' : '❌';
        const borderColor = item.passed ? '#10b981' : '#ef4444';
        return `

            
${icon} ${escapeHtml(item.criterion)}

            ${item.note ? `
💬 ${escapeHtml(item.note)}
` : ''}
        
`;
    }).join('');
    container.innerHTML = `
KI-Feedback:
${items}
        
Note wird von der Lehrkraft gesetzt.
`;
    container.style.display = '';
}

function escapeHtml(str) {
    return str.replace(/&/g,'&').replace(//g,'>').replace(/"/g,'"');
}

// Smooth scroll to element on page load if hash is present
window.addEventListener('DOMContentLoaded', function() {
    if (window.location.hash) {
        const element = document.querySelector(window.location.hash);
        if (element) {
            setTimeout(function() {
                element.scrollIntoView({ behavior: 'smooth', block: 'start' });
            }, 100);
        }
    }
});
